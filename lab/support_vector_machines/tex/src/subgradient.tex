\begin{equation*}
\begin{aligned}
f(x) &= \max(0, x) \\
\nabla f(x) &= \begin{cases}
    1,& \text{if } x\geq 0\\
    0,& \text{if} x < 0
\end{cases}
\end{aligned}
\end{equation*}
